Algoritmo "Impares_entre_10_e_20"

VAR
   it: inteiro;

INICIO
   it <- 11;
   
   escreva("Esses sao os numeros impares entre 10 e 20: ");

   ENQUANTO it < 20 FACA
            escreva(it);
            it <- it + 2;
   FIMENQUANTO
FIMALGORITMO
